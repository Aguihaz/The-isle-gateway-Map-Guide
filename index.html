Buscar Coordenada do Jogo

// ======================================================== // 🛠️ SISTEMA DE CALIBRAÇÃO REESTRUTURADO // ======================================================== // Dados base coletados const jogoA = { x: 300, y: -425 }; const pixelA = { lng: 170, lat: 316 }; const jogoB = { x: -550, y: 425 }; const pixelB = { lng: 1742, lat: 2150 }; // Multiplicador de escala corrigido para evitar distorção de eixos na imagem quadrada const escalaX = (pixelB.lng - pixelA.lng) / (jogoB.x - jogoA.x); const escalaY = (pixelB.lat - pixelA.lat) / (jogoB.y - jogoA.y); // Função de conversão jogo -> pixel Leaflet function converterJogoParaMapa(jogoX, jogoY) { const lng = pixelA.lng + (jogoX - jogoA.x) * escalaX; const lat = pixelA.lat + (jogoY - jogoA.y) * escalaY; return [lat, lng]; } // Camada global para segurar o marcador buscado e não acumular vários na tela let marcadorBusca = null; function spawnMarcador(jogoX, jogoY, texto) { const pos = converterJogoParaMapa(jogoX, jogoY); if (marcadorBusca) { map.removeLayer(marcadorBusca); } marcadorBusca = L.marker(pos).addTo(map).bindPopup(texto).openPopup(); map.setView(pos, 2); // Dá zoom automático na área do marcador } // Função chamada pelo botão "Ir" function buscarCoordenada() { const inputVal = document.getElementById("game-coords-input").value.trim(); if (!inputVal) return; // Limpa espaços extras e aceita divisão tanto por vírgula quanto por espaço simples const numeros = inputVal.replace(/,/g, ' ').split(/\s+/).map(Number).filter(n => !isNaN(n)); if (numeros.length >= 2) { const x = numeros[0]; const y = numeros[1]; spawnMarcador(x, y, `Marcador em: ${x}, ${y}`); } else { alert("Por favor, digite pelo menos dois números válidos (X e Y)."); } }
